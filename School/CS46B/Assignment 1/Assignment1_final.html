public class Vector implements Comparable
{
   private double x, y;
   
   public Vector (double thisX, double thisY)
   {
      x = thisX;
      y = thisY;
   }
   
   public double getX()
   {
      return x;
   }
   
   public double getY()
   {
      return y;
   }
   
   public String toString()
   {
      return "(x=" + x + ", y=" + y + ")";
   }
   
   public Vector add(Vector otherVector)
   {
      return new Vector(x+otherVector.x, y+otherVector.y);
   }
   
   public double getMagnitude()
   {
      return Math.sqrt(x*x + y*y);
   }
public int compareTo(Object other)
{
Vector v1 = (Vector)other;
if (this.getMagnitude() > v1.getMagnitude())
{
 return 1;
}
else if (v1.getMagnitude() > this.getMagnitude())
{
return -1;
}
else return 0;
}

import java.util.ArrayList;
import java.util.Collections;


public class VectorSortTester {
   public static void main(String[] args)
   {
      ArrayList<Vector> vectorList = new ArrayList<Vector>();
      vectorList.add(new Vector(4, 5));
      vectorList.add(new Vector(-5, -5));
      vectorList.add(new Vector(0, -1));
      vectorList.add(new Vector(3, 5));
      vectorList.add(new Vector(4, 4));

      Collections.sort(vectorList);
      System.out.println(vectorList);
      System.out.println("Expected:  [(x=0.0, y=-1.0), "
              + "(x=4.0, y=4.0), (x=3.0, y=5.0), "
              + "(x=4.0, y=5.0), (x=-5.0, y=-5.0)]");
   }
}
