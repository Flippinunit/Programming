import java.util.ArrayList;
import java.util.Comparator;

/**
 * An improved, infinite dimensional version of Hw 1's Vector.
 * (Any non-specified values are 0.)
 * @author Taylor
 */
public class Vector2C extends ArrayList<Double>
{
   /**
    * Convenience constructor for 2-D vectors.
    * @param x  First dimension value.
    * @param y  Second dimension value.
    */
   public Vector2C (double x, double y)
   {
      this.add(x);
      this.add(y);
   }
   
    /**
    * Convenience constructor for 3-D vectors.
    * @param x  First dimension value.
    * @param y  Second dimension value.
    * @param z  Third dimension value.
    */
   public Vector2C (double x, double y, double z)
   {
      this.add(x);
      this.add(y);
      this.add(z);
   }
public Vector2C()
{
}
      
      public double getMagnitude()
{
   double total = 0;
for (int i = 0; i <= this.size()-1; i++) {
   total += Math.pow(this.get(i), 2);
}
return Math.sqrt(total);
}
